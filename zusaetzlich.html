„Spread“-Operator

Kurzer Erklärartikel zu „Spread“-Operator mit Beispiel.

Der Spread-Operator in JavaScript wurde mit ECMAScript 6 (ES6) eingeführt und ermöglicht das Verteilen von Elementen eines Iterables (wie Arrays, Strings oder Objekte) auf eine andere Iterable oder einen Funktionsaufruf. Durch die Verwendung von drei Punkten "..." gefolgt von einem Ausdruck oder einer Iterable wird der Spread-Operator markiert. Dieser Operator stellt ein leistungsstarkes Werkzeug dar, das eine präzise und flexible Möglichkeit bietet, mit Daten in JavaScript zu arbeiten.

Was kann man genau mit dem Spread-Operator machen?

  • Man kann Arrays kopieren und aneinander hängen
  • Objekte miteinander kombinieren
  • Nordlist in Array umwandeln
  • String in Array verteilen

<!DOCTYPE html>
<html lang="en">

<head>
<meta charset="UTF-8">

<title>Spread-Operator</title>



</head>

<body>
<div id="komp1"></div>
<div id="komp2"></div>
<script>
// Arrays kopieren und aneinander hängen

let profs = [590120, 234030, 238324, 238235];
let studenten = [909999, 909955, 909944];

let hochschulangehoeriger = [393923, 393838, profs, studenten]
let hochschulangehoeriger2 = [393923, 393838, ...profs, ...studenten]


console.log(hochschulangehoeriger);
// Ausgabe: [393923, 393838, Array(4), Array(3)]

console.log(hochschulangehoeriger2);
// Ausgabe: [393923, 393838, 590120, 234030, 238324, 238235, 909999, 909955, 909944]

// ---------------------------------------

// Objekte kompenieren

let namen = { vorname: "Jan", nachname: "Mustermann" };
let anschrift = { strasse: "Meckenheimerstr", hausnummer: 11 };

let mitarbeiter = { pers_nr: 888888, namen, anschrift }
let mitarbeiter2 = { pers_nr: 888888, ...namen, ...anschrift }


console.log(mitarbeiter);
// Ausgabe: {pers_nr: 888888, namen: {…}, anschrift: {…}}

console.log(mitarbeiter2);
// Ausgabe: {pers_nr: 888888, vorname: 'Jan', nachname: 'Mustermann', strasse: 'Meckenheimerstr', hausnummer: 11}

// ---------------------------------------
// Nordlist in Array umwandeln 

let name = { vorname: "Jan", nachname: "Mustermann" };
console.log(name);

const elements = document.querySelectorAll("div");

console.log(elements);

// Ausgabe:  NodeList(2) [div#komp1, div#komp2]

console.log([...elements]);
// Ausgabe: [div#komp1, div#komp2]

// ---------------------------------------

// String in einem Array Verteilen

const str = "Hello";

const charArray = [...str];
console.log(charArray);
// Ausgabe: ['H', 'e', 'l', 'l', 'o']

</script>
</body>
